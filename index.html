// Define a constructor function for creating Person objects function Person(firstName, lastName) { this.firstName = firstName; this.lastName = lastName; } // Add a method to the Person prototype Person.prototype.getFullName = function () { return this.firstName + ' ' + this.lastName; }; // Create instances of the Person object const person1 = new Person('John', 'Doe'); const person2 = new Person('Alice', 'Smith'); // Access and display the full names using the getter method console.log(person1.getFullName()); // Output: John Doe console.log(person2.getFullName()); // Output: Alice Smith
